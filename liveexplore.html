04 / How to Update

Edit livedata.json

Place this file next to your HTML files. Change any value — refresh the page — done. No code needed.

  livedata.json — full reference
// ── Every value on this page + the homepage comes from here ──
{
  "courses"           : 50,       // hero "50+ Courses"
  "members"           : 5200,     // hero "5.2k Students"
  "projects"          : 120,      // hero "120+ Projects"
  "components"        : 300,      // stats band "300+ Components"
  "uptime"            : "98.7%", // dashboard panel + uptime banner

  "course_completion" : 75,       // blue ring %
  "project_success"   : 80,       // green ring %

  "bar_embedded"      : 88,       // Embedded Systems bar %
  "bar_iot"           : 74,       // IoT Agriculture bar %
  "bar_robotics"      : 61,       // Robotics bar %

  "members_list": [
    { "name": "Kavi M",  "role": "Founder",   "badge": "Admin",    "joined": "Day 1"   },
    { "name": "Arjun S", "role": "Developer", "badge": "Pro",      "joined": "3d ago"  },
    // badge options: "Admin"/"Pro" → green · "Advanced"/"IoT" → orange · "Instructor" → purple
    { "name": "...",    "role": "Student",   "badge": "Beginner", "joined": "1w ago"  }
  ]
}
